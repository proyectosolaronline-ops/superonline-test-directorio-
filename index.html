let gpsPollingInterval=null;

let lastLat=null,lastLng=null;

function refreshGPS(){

  if(!map)return;

  if(!navigator.geolocation){alert('GPS no disponible');return;}

  

  // Mostrar indicador visual

  var btn=document.querySelector('button[onclick="refreshGPS()"]');

  if(btn){btn.style.transform='rotate(360deg)';btn.style.transition='transform .6s';}

  setTimeout(function(){if(btn)btn.style.transform='rotate(0deg)';},600);

  

  navigator.geolocation.getCurrentPosition(

    function(pos){

      var newLat=pos.coords.latitude;

      var newLng=pos.coords.longitude;

      

      // Si cambió la ubicación por más de 10 metros

      if(lastLat===null||haversineKm(lastLat,lastLng,newLat,newLng)>0.01){

        userLat=newLat;

        userLng=newLng;

        

        // Actualizar círculo de ubicación

        map.eachLayer(function(l){

          if(l instanceof L.CircleMarker&&l.getTooltip()&&l.getTooltip().getContent()==='Tu ubicación'){

            l.setLatLng([userLat,userLng]);

          }

        });

        

        // Filtrar tiendas cercanas

        nearbyStores=ALL_STORES.filter(function(s){

          var cfg=s.config||{};

          var lat=parseFloat(cfg.locationLat),lng=parseFloat(cfg.locationLng);

          return !isNaN(lat)&&!isNaN(lng)&&haversineKm(userLat,userLng,lat,lng)<=RADIO_KM;

        });

        

        // Actualizar marcadores

        map.eachLayer(function(l){

          if(l instanceof L.Marker&&l.getTooltip){

            var tt=l.getTooltip();

            if(tt&&tt.getContent()!=='Tu ubicación')map.removeLayer(l);

          }

        });

        addMarkersToMap(nearbyStores);

        

                // GPS actualizado silenciosamente

      }

      lastLat=newLat;

      lastLng=newLng;

    },

    function(){},

    {enableHighAccuracy:true,timeout:5000}

  );

}

function startGPSPolling(){

  if(gpsPollingInterval)return;

  refreshGPS();

  gpsPollingInterval=setInterval(refreshGPS,3000);

}

function stopGPSPolling(){

  if(gpsPollingInterval){

    clearInterval(gpsPollingInterval);

    gpsPollingInterval=null;

  }

}

async function refreshData(){

  try{

    const [r1,r2]=await Promise.allSettled([

         fetch(BACKEND+'?action=stores.directory&t='+Date.now()),

      fetch(BACKEND_MAESTRO+'?action=stores.directory&t='+Date.now())    ]);

    var sl=[],se=[];

    if(r1.status==='fulfilled'){const d=await r1.value.json();sl=(d.stores||[]).filter(s=>!s.isExternal);}

    if(r2.status==='fulfilled'){const d=await r2.value.json();se=(d.stores||[]).filter(s=>s.isExternal===true);}

    ALL_STORES=[...sl,...se].map(s=>{s.config=typeof s.config==='string'?JSON.parse(s.config):(s.config||{});return s;});

    

    // Refiltrar cercanas

    if(userLat&&userLng){

      nearbyStores=ALL_STORES.filter(s=>{

        var cfg=s.config||{};

        var lat=parseFloat(cfg.locationLat),lng=parseFloat(cfg.locationLng);

        return !isNaN(lat)&&!isNaN(lng)&&haversineKm(userLat,userLng,lat,lng)<=RADIO_KM;

      });

    }

    

    // Cerrar popups abiertos

    map.closePopup();

    

    // Actualizar marcadores

function actualizarSchemaNearbyStores() {

  try {

    const schemaEl = document.getElementById('schema-nearby-stores');

    const items = nearbyStores.map((s, i) => {

function obtenerUbicacionGPS() {

  var statusEl = document.getElementById('contacto-gps-status');

  var inputEl = document.getElementById('contacto-ubicacion');

  

  if (!navigator.geolocation) {

    statusEl.style.display = 'block';

    statusEl.style.color = 'red';

    statusEl.textContent = '❌ GPS no disponible en este dispositivo';

    return;

  }

  

  statusEl.style.display = 'block';

  statusEl.style.color = '#1a73e8';

  statusEl.textContent = '⏳ Obteniendo ubicación...';

  

  navigator.geolocation.getCurrentPosition(

    function(pos) {

      var lat = pos.coords.latitude.toFixed(6);

      var lng = pos.coords.longitude.toFixed(6);

      inputEl.value = lat + ',' + lng;

      statusEl.style.color = '#16a34a';

      statusEl.textContent = '✅ Ubicación obtenida: ' + lat + ', ' + lng;

      setTimeout(function() { statusEl.style.display = 'none'; }, 3000);

    },

    function(err) {

      statusEl.style.color = 'red';

      statusEl.textContent = '❌ No se pudo obtener. Escribe las coordenadas manualmente.';

    },

    { enableHighAccuracy: true, timeout: 10000 }

  );

}

  var localidad = document.getElementById('contacto-localidad').value.trim();

  var estado = document.getElementById('contacto-estado').value.trim();

  var link = document.getElementById('contacto-link').value.trim();

  var ubicacion = document.getElementById('contacto-ubicacion').value.trim();  

   var foto = document.getElementById('contacto-foto').value.trim();

  if (!nombre || !tienda || !telefono || !tipo || !localidad) {

    alert('Completa todos los campos requeridos (*)');

    return;

  }

  

  if (telefono.length < 10) {

    alert('Ingresa un WhatsApp válido de 10 dígitos');

    return;

  }

  

  try {

    var resp = await fetch(BACKEND_MAESTRO, {

      method: 'POST',

      headers: {'Content-Type': 'text/plain;charset=utf-8'},

      body: JSON.stringify({

        action: 'solicitud.contacto',

        nombre_cliente: nombre,

        nombre_tienda: tienda,

        telefono: telefono,

        tipo_negocio: tipo,

        colonia: colonia,

        localidad: localidad,

        estado: estado || 'Querétaro',

        link_negocio: link,

        ubicacion: ubicacion,

        foto: foto,

         movil: ubicacion ? true : false      })

    });

    

    var data = await resp.json();

    

    if (data.ok) {

      alert('✅ Solicitud enviada correctamente. Te contactaremos pronto.');

      cerrarFormularioContacto();

    } else {

      alert('❌ Error: ' + (data.error || 'Intenta de nuevo'));

    }

  } catch(e) {

    alert('❌ Error de conexión. Intenta de nuevo.');

  }

}
